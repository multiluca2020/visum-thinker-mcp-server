Examples

-
  • Python 
# for zones
zone1 = Visum.Net.Zones.ItemByKey(100)
zone2 = Visum.Net.Zones.ItemByKey(200)
ZoneElements = Visum.CreateNetElements()
ZoneElements.Add(zone1)
ZoneElements.Add(zone2)
RouteSearch = Visum.Analysis.RouteSearchPrT
RouteSearch.Clear()
RouteSearch.Execute(NetElms=ZoneElements, TSysCode='C', searchCriterion=2)
node_chain = RouteSearch.NodeChainPrT
Visum.Net.AddPathSet(1)
Visum.Net.AddPath(no=1, PathSet=1, fromZone=zone1, toZone=zone2, PathItems=node_chain)

# for nodes
node1 = Visum.Net.Nodes.ItemByKey(10)
node2 = Visum.Net.Nodes.ItemByKey(40)
NodeElements = Visum.CreateNetElements()
NodeElements.Add(node1)
NodeElements.Add(node2)
RouteSearch.Clear()
RouteSearch.Execute(NetElms=NodeElements, TSysCode='C', searchCriterion=2)
node_chain = RouteSearch.NodeChainPrT
Visum.Net.AddPath(no=2, PathSet=1, fromZone=node1, toZone=node2, PathItems=node_chain
