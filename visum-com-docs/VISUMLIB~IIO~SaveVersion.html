#-------------------------------------------------------------------------------
# This example create a VISUM-Object and load a version file.
# The length of all links , V0 for privat transport and the time for the
# transportsystem bus are modified an the result will be saved in a new version file.
#-------------------------------------------------------------------------------
import platform
import win32com.client
import os, sys

# create the Visum-Object (connect the variable Visum to the software VISUM)
Visum = win32com.client.gencache.EnsureDispatch('Visum.Visum')

versionPath = os.path.join(os.path.dirname(sys.argv[0]), r"..\\..\\network","Example.ver")

# load version
Visum.IO.LoadVersion(versionPath)

# go over all links
AllLinks = Visum.Net.Links.GetAll
for link in AllLinks:
    # get the length in meter
    # then write new length
    Length = link.AttValue("LENGTH")
    Length = Length * 1.1
    link.SetAttValue("Length", Length)

# save the version
saveVersionPath = os.path.join(os.path.dirname(sys.argv[0]), r"..\..\Network", "EXAMPLE_SaveVersion.VER")
Visum.IO.SaveVersion(saveVersionPath)

# close Visum
AllLinks= None
link = None
Visum = None
