#-------------------------------------------------------------------------------
# This example shows how to create a VISUM-Object and load a version file.
#-------------------------------------------------------------------------------
import platform
import win32com.client
import os, sys

# create the Visum-Object (connect the variable Visum to the software VISUM)
Visum = win32com.client.gencache.EnsureDispatch('Visum.Visum')


if platform.win32_ver()[1].startswith('5'):
    public = os.environ["ALLUSERSPROFILE"]
else :
    public = os.environ["PUBLIC"]

versionPath = os.path.join(os.path.dirname(sys.argv[0]), r"..\\..\\network","Example.ver")

# load version
Visum.IO.LoadVersion(versionPath)

# Close Visum
Visum = None
