myOperation = Visum.Procedures.Operations.ItemByKey(5)
myOperation.DemandMatrixCorrectionParameters.AttValue("PROCEDUREVARIANT")

# This returns the answer:
# 'TFLOWFUZZY'

# The parameter "Max. correction factor" can be changed on the tab "Configuration" of the interactive dialog.
# The COM object offers a similar division:

MyTFlowFuzzyConfiguration = myOperation.DemandMatrixCorrectionParameters.ConfigParameters
MyTFlowFuzzyConfiguration.AttValue("TFLOWFUZZYMAXCORRECTFACTOR")

# This returns the answer:
# 20.0

# Change the value with:
MyTFlowFuzzyConfiguration.SetAttValue("TFLOWFUZZYMAXCORRECTFACTOR",15)
