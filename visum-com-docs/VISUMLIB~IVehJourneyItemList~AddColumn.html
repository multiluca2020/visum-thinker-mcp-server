Summary
This method adds a new column to the end (in the beginning) of the empty list and defines its format.
Syntax
Visual Basic
Public Sub AddColumn( _
   ByVal Attribut As String, _
   Optional ByVal Format As StringFormatTypeT = defaultfmt, _
   Optional ByVal DecPlaces As Integer = -1, _
   Optional ByVal DisplayUnits As Boolean = False, _
   Optional ByVal GroupOrAggrFunction As AggregationFunction = AggregationFunctionDefault, _
   Optional ByVal WeightedAvgAttribute As String = "" _
)

Examples


-
  • Python 
# creates list of links
linkList = Visum.Lists.CreateLinkList
#opens list of links
linkList.Show()
#add columns to link list
linkList.AddColumn(Attribut="FromNodeNo", GroupOrAggrFunction=1) #group by this column
linkList.AddColumn(Attribut="VolVehPrT(AP", GroupOrAggrFunction=4, decplaces=2) #aggregate = average
linkList.AddColumn(Attribut="V0PRT", GroupOrAggrFunction=3, DisplayUnits=True) # max speed with unit
