# For a long explanation instructing on how to find this out with the COM help and the Attributes List:

# This is about the setting
# Graphics -> Edit graphic parameters -> Links -> Draw layer links -> Draw links

# See: Lists -> Attributes
# Choose that it does not only show "Only network object types" (its default) but "All attributes".
# Filter the column name ObjTypeName for something with "links..." and you can find "LinkGraphicParameters". It leads to an Attribute called "DRAW".
 
myNetEditor2DGPar = Visum.Net.GraphicParameters
myLinkGPar = myNetEditor2DGPar.Links

# Ask for its current value:
myLinkGPar.AttValue("DRAW")
# The answer could be: 1.0
# Set the value:
myLinkGPar.SetAttValue("DRAW", 1)
